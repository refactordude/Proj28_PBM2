def get_summary_route(
    request: Request,
    platform_id: Annotated[str, Path(pattern=PLATFORM_ID_PATTERN, min_length=1, max_length=128)],
    x_regenerate: Annotated[str | None, Header()] = None,

    target_id = _resolve_target_id(hx_target, platform_id)

    regenerate = (x_regenerate or "").lower() == "true"   # ← canonical parsing; ONLY the literal "true" triggers
