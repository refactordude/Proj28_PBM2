```python
@router.get("/ask", response_class=HTMLResponse)
def ask_page(request: Request, response: Response):
    """Chat shell. No starter chips (D-CHAT-10). LLM dropdown verbatim from Phase 6 (D-CHAT-11)."""
    settings = request.app.state.settings
    active_llm = resolve_active_llm(settings, request)

    _ensure_session_cookie(request, response)

    return templates.TemplateResponse(
        request,

        {
            "page_title": "Ask",
            "active_llm": active_llm,
            "available_llms": available_llms,
        },
    )


def _ensure_session_cookie(request: Request, response: Response) -> str:
    """Set pbm2_session cookie on first visit (RESEARCH Gap 6 — uuid4.hex, Path=/, SameSite=Lax,
    HttpOnly=True, Secure=False — same shape as pbm2_llm cookie per Pitfall 8)."""
    sid = request.cookies.get(_PBM2_SESSION_COOKIE)
    if sid:
        return sid
    sid = uuid.uuid4().hex
    response.set_cookie(
        _PBM2_SESSION_COOKIE,
        sid,
        max_age=31536000,
        httponly=True,
        samesite="lax",
        secure=False,
        path="/",
    )
    return sid
```

**`GET /ask/stream/{turn_id}` handler** (async def — REQUIRED for SSE per RESEARCH Pitfall 1):

```python
@router.get("/ask/stream/{turn_id}")
async def ask_stream(turn_id: str, request: Request):
    """SSE endpoint. ASYNC def is REQUIRED here — the v2.0 Phase 6 sync-only invariant
    is narrowed in plan 05 to allow async on /ask/stream and /ask/cancel only.
    """
    settings = request.app.state.settings

    # Authentication: turn_id must belong to the requesting session (T-03-04-01)
    request_sid = request.cookies.get(_PBM2_SESSION_COOKIE) or ""
    try:
        owner_sid = get_session_id_for_turn(turn_id)
    except KeyError:
        return Response(status_code=404)
    if owner_sid != request_sid:
        return Response(status_code=403)

    llm_cfg = resolve_active_llm(settings, request)
    if llm_cfg is None:
        async def _no_llm():
            yield ServerSentEvent(
                event="error",
                data=templates.get_template("ask/_error_card.html").render(
                    severity="hard", reason="unconfigured",
                    heading="Something went wrong.",
                    body="No LLM backend is configured.",
                    detail="", original_question="",
                ),
            )
        return EventSourceResponse(_no_llm(), background=BackgroundTask(pop_turn, turn_id))

    db_adapter = request.app.state.db_adapter  # the project's existing app.state DB handle
    deps = ChatAgentDeps.model_construct(
        db=db_adapter,
        agent_cfg=settings.app.agent,
        active_llm_type=llm_cfg.type,
    )
    agent = build_chat_agent(_resolve_pydantic_ai_model(llm_cfg))
    cancel_event = get_cancel_event(turn_id)
    question = get_pending_question(turn_id)
    msg_history = get_session_history(owner_sid, limit=12)

    # On-completion callback: persist new messages with D-CHAT-11 path scrub
    def _persist(new_messages):
        append_session_history(owner_sid, new_messages, active_llm_type=llm_cfg.type)

    async def event_generator():
        async for ev in stream_chat_turn(
            agent=agent,
            deps=deps,
            question=question,
            message_history=msg_history,
            cancel_event=cancel_event,
            chat_max_steps=settings.app.agent.chat_max_steps,
            rejection_cap=5,
            on_run_complete=_persist,
        ):
            # For final events, re-render with table_html + chart_html populated by
            # executing PresentResult.sql against the DB (D-CHAT-05 row table is mandatory)
            if ev["event"] == "final":
                ev = {"event": "final", "html": _hydrate_final_card(ev["html"], deps)}
            yield ServerSentEvent(event=ev["event"], data=ev["html"])

    return EventSourceResponse(
        event_generator(),
        background=BackgroundTask(pop_turn, turn_id),
        ping=15,
    )
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → `/ask/stream/{turn_id}` | An attacker who guesses or steals a turn_id could read another user's SSE stream (their thoughts, SQL, results) — primary risk for a multi-tenant intranet. |
| Browser → `/ask/cancel/{turn_id}` | Same as above — an attacker could cancel another user's turn. |
| Agent string → DOM (Jinja autoescape) | Agent-generated thought text, tool-call args, tool-result content all flow into HTML; without `| e` autoescape an LLM-controlled string could inject `<script>`. |
| Plotly chart HTML → DOM (`| safe`) | `_final_card.html` uses `| safe` on `chart_html` (server-rendered Plotly); Plotly bundle defines its own DOM injection — trusted server-side render. |
