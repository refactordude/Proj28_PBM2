    ```python
    """Export dialog — Excel + CSV downloads for Browse page (EXPORT-01, EXPORT-02).

    Opens as @st.dialog triggered from the Pivot tab's Export button. Captures the
    currently-visible wide pivot (D-15) or the raw long-form fetch_cells result (D-16
    power-user scope).
    """
    from __future__ import annotations

    import io
    import re
    from datetime import datetime
    from typing import Optional

    import pandas as pd
    import streamlit as st

    _EXCEL_MIME = "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"
    _CSV_MIME = "text/csv"

    _FMT_XLSX = "Excel (.xlsx)"
    _FMT_CSV = "CSV (.csv)"
    _SCOPE_WIDE = "Current view (wide)"
    _SCOPE_LONG = "Full result (long)"


    def _sanitize_filename(name: str) -> str:
        """Path-traversal-safe filename. Strips '..', non-[A-Za-z0-9_-.] chars,
        collapses underscores, strips leading/trailing dots/underscores, truncates to 128.
        Returns 'ufs_export' when sanitization removes everything."""
        if name is None:
            return "ufs_export"
        # 1. Remove '..' occurrences (defense against ../ path traversal)
        s = name.replace("..", "")
        # 2. Remap any character outside the allowed set to '_'
        s = re.sub(r"[^A-Za-z0-9_\-.]", "_", s)
        # 3. Collapse repeated underscores
        s = re.sub(r"_+", "_", s)
        # 4. Strip leading/trailing dots and underscores
        s = s.strip("._")
        # 5. Truncate
        s = s[:128]
        return s or "ufs_export"


    def _default_filename(scope_token: str, ext: str) -> str:
        """Build the default filename 'ufs_{scope}_{YYYYMMDD}.{ext}'."""
        date = datetime.now().strftime("%Y%m%d")
        return f"ufs_{scope_token}_{date}.{ext}"


    def _write_excel_bytes(df: pd.DataFrame, sheet_name: str = "UFS") -> bytes:
        """Write df to an in-memory xlsx via openpyxl. Auto-size columns by header length."""
        import openpyxl.utils  # local import to keep module import cheap

        buf = io.BytesIO()
        with pd.ExcelWriter(buf, engine="openpyxl") as writer:
            df.to_excel(writer, index=False, sheet_name=sheet_name)
            worksheet = writer.sheets[sheet_name]
            for idx, column_name in enumerate(df.columns):
                header_len = len(str(column_name))
                width = min(50, max(12, header_len))
                letter = openpyxl.utils.get_column_letter(idx + 1)
                worksheet.column_dimensions[letter].width = width
        return buf.getvalue()


    def _write_csv_bytes(df: pd.DataFrame) -> bytes:
        """Write df to CSV with utf-8-sig BOM (enables Excel double-click). index=False.

        NOTE: to_csv(encoding=...) returns a str without BOM when no path is given,
        so we encode once with utf-8-sig to add exactly one BOM. Do NOT pass
        encoding='utf-8-sig' to to_csv AND then .encode('utf-8-sig') — that yields a
        double BOM and corrupts the first cell in Excel.
        """
        return df.to_csv(index=False).encode("utf-8-sig")


    @st.dialog("Export data")
    def render_export_dialog(
        df_wide: Optional[pd.DataFrame],
        df_long: Optional[pd.DataFrame],
    ) -> None:
        """Render the export dialog. At least one DataFrame must be non-empty."""
        wide_ok = df_wide is not None and not df_wide.empty
        long_ok = df_long is not None and not df_long.empty
        if not wide_ok and not long_ok:
            raise ValueError("render_export_dialog: at least one of df_wide or df_long must be non-empty")

        # Format radio
        fmt = st.radio("Format", [_FMT_XLSX, _FMT_CSV], horizontal=True, key="export.format")
        ext = "xlsx" if fmt == _FMT_XLSX else "csv"

        # Scope radio — only when both scopes are available
        if wide_ok and long_ok:
            scope = st.radio("Scope", [_SCOPE_WIDE, _SCOPE_LONG], index=0, key="export.scope")
        elif wide_ok:
            scope = _SCOPE_WIDE
            st.caption(f"Scope: {_SCOPE_WIDE}")
        else:
            scope = _SCOPE_LONG
            st.caption(f"Scope: {_SCOPE_LONG}")

        scope_token = "wide" if scope == _SCOPE_WIDE else "long"
        scope_df = df_wide if scope == _SCOPE_WIDE else df_long  # type: ignore[assignment]

        # Filename input with default template
        default_name = _default_filename(scope_token, ext)
        filename = st.text_input("Filename", value=default_name, key="export.filename")
        st.caption("File will be saved to your Downloads folder.")

        # Build file bytes EAGERLY so st.download_button is primed on first render (Pitfall 6)
        base_stem = filename.rsplit(".", 1)[0] if "." in filename else filename
        safe_stem = _sanitize_filename(base_stem)
        full_name = f"{safe_stem}.{ext}"

        if fmt == _FMT_XLSX:
            file_bytes = _write_excel_bytes(scope_df)
            mime = _EXCEL_MIME
        else:
            file_bytes = _write_csv_bytes(scope_df)
            mime = _CSV_MIME

        # Download + Close buttons
        col_dl, col_close = st.columns([1, 1])
        with col_dl:
            st.download_button(
                "Download",
                data=file_bytes,
                file_name=full_name,
                mime=mime,
                type="primary",
                key="export.download",
            )
        with col_close:
            if st.button("Close", type="secondary", key="export.close"):
                st.rerun()
    ```

| st.download_button bytes -> browser | one-way download; no read-back surface |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-01 | Tampering | path traversal via user-typed filename (`../../etc/passwd.xlsx`) | mitigate | `_sanitize_filename` strips `..` and remaps path separators (`/`, `\`) to `_` BEFORE use; the returned string is pure basename with no path components; st.download_button passes this as `file_name=` which the browser interprets as a filename hint only (browser save-as dialog is the ultimate gate, but sanitization gives defense in depth) |
| T-07-02 | Tampering | XSS via filename containing HTML (`<script>`) | mitigate | charset clamp to `[A-Za-z0-9_\-.]` removes all angle brackets, quotes, ampersands — file_name is pure ASCII-alphanumeric plus dot/dash/underscore |
| T-07-03 | DoS | user exports 100k-row DF to xlsx | mitigate | D-07 30-column cap + DATA-07 200-row cap already cap the upstream df_wide; long-form scope is bounded by fetch_cells' 200-row LIMIT; worst case is ~200 rows × 30 cols xlsx, well under any memory threshold |
| T-07-04 | Information Disclosure | exporting unfiltered data by accident | accept | D-15 contract makes exports reflect what the user sees in the UI — no fresh fetch, no superset; the Scope radio is an explicit opt-in to long-form |
| T-07-05 | Information Disclosure | session_state leaks df_wide across browser tabs | accept | Streamlit's session_state is per-session-per-tab; no cross-tab leak by design |
| T-07-06 | Repudiation | user downloads data and no audit log exists | accept | Phase 1 is intranet read-only; audit logging is a pre-deployment concern tracked alongside auth (FOUND-01 deferral) |
