```python
def _write_fm(content_dir: Path, pid: str, **kwargs) -> Path:
    target = content_dir / f"{pid}.md"
    lines = ["---"]
    for k, v in kwargs.items():
        lines.append(f"{k}: {v}")
    lines.append("---")
    lines.append("")
    lines.append("# body")
    target.write_text("\n".join(lines), encoding="utf-8")
    return target
```

```python
import urllib.parse
def _post_form_pairs(client, url: str, pairs: list[tuple[str, str]]) -> httpx.Response:
    body = urllib.parse.urlencode(pairs)
    return client.post(url, content=body, headers={"Content-Type": "application/x-www-form-urlencoded"})
```
